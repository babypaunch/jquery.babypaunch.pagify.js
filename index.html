var config = {
	current: 1 //현재 페이지 번호, offset은 1부터, 필수
	, total: 0 //전체 건수, 필수
	, pager: 10 //노출할 pager 갯수
	, isText: true //pager의 스타일을 결정, true이면 숫자링크, false면 input으로 출력
	, combo: [10, 20, 30, 50] //listing group
	, list: 20 //한 화면에 listing될 갯수를 지정.
	, showListed: true//전체 건수 show/hide
	, showCombo: true //콤보박스 show/hide
	, showPrevNext: true //이전/다음 버튼 show/hide
	, showFirstEnd: true //처음/끝 버튼 show/hide
	, attr: "index" //set custom attr name
	, currentClass: "on" //현재 페이지에 해당하는 번호에 class를 추가
	, html: { //pager로 보여질 dom 객체를 jquery 방식으로 주입
		total: $("<span>")
		, first: $("<a>", {"href": "#", "text": " ≪ "})
		, prev: $("<a>", {"href": "#", "text": " ＜ "})
		, num: $("<a>", {"href": "#"})
		, next: $("<a>", {"href": "#", "text": " ＞ "})
		, end: $("<a>", {"href": "#", "text": " ≫ "})
	}
}; //end: var config = {

위의 코드는 pagify를 사용하기 위한 기본 설정값으로, 필수로 설정해야할 current와 total값을 제외하고는 모두 상황에 따라 추가해서 사용하면 된다. config를 염두해두고 아래의 2가지 예제를 통해서 사용법을 알아보자. 

ex1)

$("#pager1")
	.pagify({"current": 5 //현재 페이지의 번호를 항상 받아야 함.
		, "total": 999 //전체 건수를 항상 받아야함.
		, "combo": [20, 40, 60, 80, 100] //default가 아닌 다른 값으로 select box의 값을 지정했다.
		, "list": 40 //한페이지 기본으로 보여줄 리스트 갯수를 변경했다.
		, "html": {
			//전체 건수를 출력하는 태그와 스타일(class 이용을 권장함)을 지정함함.
			"total": $("", {"style": "color: red;"}) 
			//'처음'버튼의 태그와 몇몇 속성값 및 스타일(class 이용을 권장함)을 지정함.
			, "first": $("", {"id": "pager-first", "href": "#", "style": "margin: 0 3px;" 
				//'처음'버튼 내부 태그를 지정하고 스타일(class 이용을 권장함)을 지정함.
				, "html": $("", {"text": "first", "style": "background: silver; color: white; padding: 0 2px;"})
			})
			, "prev": $("", {"id": "pager-prev", "href": "#", "style": "margin: 0 3px;"
				, "html": $("", {"text": "prev", "style": "background: silver; color: white; padding: 0 2px;"})
			})
			//숫자 출력의 경우는 currentClass 값을 이용하여 text를 출력하게 강제화되어 있음.
			//아래의 스타일은 숫자 출력시 일반적인 스타일을 적용하기 위해 입력한 예시임.
			, "num": $("", {"href": "#", "style": "margin: 0 5px;"})
			, "next": $("", {"id": "pager-next", "href": "#", "style": "margin: 0 3px;"
				, "html": $("", {"text": "next", "style": "background: silver; color: white; padding: 0 2px;"})
			})
			, "end": $("", {"id": "pager-end", "href": "#", "style": "margin: 0 3px;"
				, "html": $("", {"text": "end", "style": "background: silver; color: white; padding: 0 2px;"})
			})
		}
	})
	.on("click", "a", function(e){ //pager 내의 모든 a태그를 클릭시 동작
		e.preventDefault();
		alert($(this).attr("data-index"));
	})
	.on("change", "select", function(e){ //pager 내의 select box를 변경시 동작
		alert($(this).val());
	})
;

$("#pager2")
	.pagify({"current": 23
		, "total": 999
		, "isText": false //input box 형태의 pager를 사용하기 위해 설정
		, "showCombo": false //select box를 숨김 처리함.
	})
	.on("change", "input", function(){ //pager 내의 input box의 값을 변경시 동작
		$(this).attr("data-index", Number($(this).val()) - 1);
		alert($(this).attr("data-index") + ", " + $(this).val());
	})
	.on("click", "a", function(e){ //pager 내의 모든 a태그를 클릭시 동작
		e.preventDefault();
		alert($(this).attr("data-index"));
	})
;
