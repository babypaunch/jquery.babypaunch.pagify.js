ex1)

$("#pager1")
	.pagify({"current": 23, "total": 999})
	.on("click", "a", function(){
		//sample process
		$("#pager1").pagify({"current": Number($(this).attr("data-index")) + 1, "total": 999, "list": $("#pager1 option:selected").val()});
	})
	.on("change", "select", function(){
		//sample process
		$("#pager1").pagify({"current": 1, "total": 999, "list": $(this).val()});
	})
	.find("a").css({"color": "silver"})
	.end().find("a.on").css({"color": "red", "font-weight": "bold"})
	.end().find("a.first, a.prev, a.next, a.end").css({"color": "black", "font-weight": "bold"})
;
