Example 1. - 기본 사용, 전체 111건, 현재 1페이지
$("#pager1") .pagify({"total": 111 , "current": 1 }) .on("click", "a", function(e){ e.preventDefault(); alert($(this).attr("data-pager-index")); }) .on("change", "select", function(){ alert($(this).val()); });
